def autonomous_self_fixer(task, max_attempts=3):
    print(f" Starting Mission: {task}")
    history = ""
    
    for i in range(max_attempts):
        print(f"\n Attempt {i+1} in progress...")
        
        prompt = f"""
        You are an AI Research Engineer. 
        TASK: {task}
        PAST_ATTEMPTS_HISTORY: {history}
        
        INSTRUCTION: 
        1. Write a Python script to solve the task.
        2. If history shows an error, ANALYZE why it failed and FIX it.
        3. Use synthetic data if external files are missing.
        4. Return ONLY the python code block.
        """
        
    
        ai_response = llm.invoke([
            SystemMessage(content="You are a God-mode Debugger."),
            HumanMessage(content=prompt)
        ]).content
        
       
        clean_code = ai_response.replace("```python", "").replace("```", "").strip()
        

        result = advanced_run_code(clean_code)
        
        if result["status"] == "success":
            print(f" TABAHI! AI successfully fixed itself on attempt {i+1}.")
            return result["output"]
        else:
            print(f" Galti mil gayi: {result['output'].splitlines()[-1]}")
          
            history += f"\nAttempt {i+1} failed with code:\n{clean_code}\nError: {result['output']}"
            
    return "Mission Failed: AI couldn't fix it. Human intervention needed."
